const int ledPin = 13; // Pin where the LED is connected

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  if (Serial.available() > 0) {
    char command = Serial.read();
    if (command == '1') {
      digitalWrite(ledPin, HIGH); // Turn LED on
    } else if (command == '0') {
      digitalWrite(ledPin, LOW); // Turn LED off
    }
  }
}
